# How To Make The Project
Run the script ```npx @lucas11/discord-json-cli``` and  now you are ready to configurate!


![Sample configuration](https://nsa39.casimages.com/img/2018/11/04/181104042118770870.png)

```json
{
    "general": {
        "token": "",
        "debug": false
    },
    "welcome": {
        "enabled": true,
        "type": "channel",
        "channel_id": "508650923447287812",
        "message": "Welcome {user} to {guild} ! Your id is {id} !"
    },
    "byebye": {
        "enabled": true,
        "type": "channel",
        "channel_id": "508650923447287812",
        "message": "See you soon {user} in {guild} ! His id was {id} !"
    },
    "commands": [{
            "command": "!hey",
            "action": "send_message",
            "message": "Hey!"
        },
        {
            "command": "!heymp",
            "action": "send_dm",
            "message": "Hey!"
        },
        {
            "command": "!givemyskill",
            "action": "add_role",
            "role_id": ""
        },
        {
            "command": "!removemyskill",
            "action": "remove_role",
            "role_id": ""
        },
        {
            "command": "!switchmyskill",
            "action": "switch_role",
            "role_id": ""
        }
    ],
    "reaction_messages": [{
        "channel_id": "",
        "message_id": "",
        "action": "role",
        "role_id": "",
        "reaction": "ðŸŒ "
    }],
    "presence": {
        "enabled": true,
        "type": "watching",
        "text": "discord.json <3",
        "streaming_url": "https://www.twitch.tv/thomasbysfr"
    }
}
```
